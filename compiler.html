#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX_NAME 50
#define MAX_CATEGORY 20

struct Event {
    int id;
    char name[MAX_NAME];
    char category[MAX_CATEGORY];
    char date[11];
    char time_stamp[9];
    struct Event *next;
};

struct StackNode {
    struct Event *event_data;
    struct StackNode *next;
};

struct Event *head = NULL;
struct StackNode *stack_top = NULL;
int event_count = 0;
int next_id = 1;

void addEvent() {
    struct Event *newNode = (struct Event*)malloc(sizeof(struct Event));
    if (newNode == NULL) {
        printf("Memory allocation failed!\n");
        return;
    }
    printf("Enter event name: ");
    scanf(" %[^\n]", newNode->name);
    printf("Enter category: ");
    scanf(" %s", newNode->category);
    getCurrentDate(newNode->date);
    getCurrentTime(newNode->time_stamp);
    newNode->id = next_id++;
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
    } else {
        struct Event *temp = head;
        while (temp->next != NULL)
            temp = temp->next;
        temp->next = newNode;
    }
    pushToStack(newNode);
    event_count++;
    printf("Event added successfully!\n");
}

void deleteEvent() {
    if (head == NULL) {
        printf("Timeline is empty.\n");
        return;
    }
    int id;
    printf("Enter event ID to delete: ");
    scanf("%d", &id);

    if (head->id == id) {
        struct Event *temp = head;
        head = head->next;
        free(temp);
        event_count--;
        printf("Event deleted.\n");
        return;
    }
    struct Event *prev = head, *curr = head->next;
    while (curr != NULL) {
        if (curr->id == id) {
            prev->next = curr->next;
            free(curr);
            event_count--;
            printf("Event deleted.\n");
            return;
        }
        prev = curr;
        curr = curr->next;
    }
    printf("Event not found.\n");
}

Logic Explained — Step by Step

🧠 malloc() — Memory Allocation

▼ 
struct Event *newNode = (struct Event*)malloc(sizeof(struct Event));
  1. sizeof(struct Event) calculates the total bytes needed for one Event node (id + name + category + date + time + pointer).
  2. malloc() requests that many bytes from the heap (dynamic memory).
  3. The OS finds a free block of memory and returns a void* pointer to it.
  4. We cast it to (struct Event*) so the compiler knows the type.
  5. If no memory is available, malloc returns NULL — always check for this!
malloc(88 bytes) → Heap: 0x7FF... → newNode ✓

📦 Node Creation

▼ 
newNode->id = next_id++;
strcpy(newNode->name, "Phone Unlocked");
strcpy(newNode->category, "Phone");
newNode->next = NULL;
  1. The -> operator accesses members through a pointer.
  2. Each field is filled: ID is auto-incremented, name and category from user input.
  3. Date and time are auto-generated using time() and strftime().
  4. next is set to NULL because this node is not yet connected to the list.
id: 1 name: "Phone Unlocked" next: NULL

🔗 Linked List Insertion

▼ 
if (head == NULL) {
    head = newNode;
} else {
    struct Event *temp = head;
    while (temp->next != NULL)
        temp = temp->next;
    temp->next = newNode;
}
  1. If the list is empty (head == NULL), the new node becomes the head.
  2. Otherwise, traverse from head following next pointers until the last node.
  3. The last node has next == NULL — that's how we know it's the tail.
  4. Set tail->next = newNode to append at the end.
HEAD → Node 1 → Node 2 → NEW ✓ → NULL

👉 Pointer Update

▼ 
// Before: tail->next == NULL
tail->next = newNode;
// After:  tail->next == newNode
//         newNode->next == NULL

Pointers are just memory addresses. When we write temp->next = newNode, we're storing the address of newNode in the previous node's next field. This is how linked lists "link" — each node holds the address of the next one in the chain.

0x100 → 0x200 → 0x300 (new) → NULL

⬆️ Push Operation

▼ 
void pushToStack(struct Event *event) {
    struct StackNode *sn = malloc(sizeof(struct StackNode));
    sn->event_data = malloc(sizeof(struct Event));
    memcpy(sn->event_data, event, sizeof(struct Event));
    sn->next = stack_top;
    stack_top = sn;
}
  1. Allocate a new StackNode.
  2. Deep-copy the event data (so the stack has its own copy).
  3. Point the new node's next to the current top.
  4. Update stack_top to point to the new node — it's now on top.

⬇️ Pop Operation

▼ 
struct Event* popFromStack() {
    if (stack_top == NULL) return NULL;
    struct StackNode *temp = stack_top;
    struct Event *data = temp->event_data;
    stack_top = stack_top->next;
    free(temp);
    return data;
}
  1. Save the current top node in a temp pointer.
  2. Extract the event data pointer (we'll return this).
  3. Move stack_top to the next node below.
  4. Free the StackNode wrapper (but NOT the event data — caller will use it).
  5. Return the event data to the caller.

🗑️ Deletion & free()

▼ 
prev->next = curr->next;  // bypass the node
free(curr);               // release memory
  1. Find the node to delete and its previous node.
  2. Set prev->next = curr->next — this bypasses the target node.
  3. Call free(curr) to return the memory to the OS.
  4. Special case: if deleting the head, just set head = head->next first.
prev ⤳ curr ✗ ⤳ curr→next

🔄 Traversal

▼ 
struct Event *temp = head;
while (temp != NULL) {
    printf("%s\n", temp->name);
    temp = temp->next;
}
  1. Start with a temp pointer at the head.
  2. Process the current node (print, compare, count, etc.).
  3. Move to the next node: temp = temp->next.
  4. Stop when temp == NULL — we've reached the end.
  5. This pattern is O(n) time and O(1) space — the foundation of every operation.
temp → Node 1 → Node 2 → NULL (stop)
